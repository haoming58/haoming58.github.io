```python
import torch.nn as nn

class SelfAttention_v1(nn.Module):
    def __init__(self, d_in, d_out):
        super().__init__()
        # 1. 定义三个权重矩阵 (三巨头)
        self.W_query = nn.Parameter(torch.rand(d_in, d_out))
        self.W_key   = nn.Parameter(torch.rand(d_in, d_out))
        self.W_value = nn.Parameter(torch.rand(d_in, d_out))

    def forward(self, x):
        # x 的形状: [Batch_Size, Seq_Len, d_in]
        
        # 2. 计算 Q, K, V (投影/变身)
        keys = x @ self.W_key
        queries = x @ self.W_query
        values = x @ self.W_value
        
        # 3. 计算注意力分数 (Queries x Keys^T)
        # 这里的 .transpose 是为了把 keys 转置以便矩阵相乘
        attn_scores = queries @ keys.transpose(-2, -1) 
        
        # 4. 缩放 (防止数值爆炸)
        d_k = keys.shape[-1]
        attn_scores_scaled = attn_scores / (d_k ** 0.5)
        
        # 5. 归一化 (Softmax)
        attn_weights = torch.softmax(attn_scores_scaled, dim=-1)
        
        # 6. 加权求和 (Weights x Values)
        context_vec = attn_weights @ values
        
        return context_vec
```

```python
class CausalAttention(nn.Module):
    def __init__(self, d_in, d_out, context_length, dropout, qkv_bias=False):
        super().__init__()
        self.d_out = d_out
        # 1. 定义 Q, K, V 投影层
        self.W_query = nn.Linear(d_in, d_out, bias=qkv_bias)
        self.W_key   = nn.Linear(d_in, d_out, bias=qkv_bias)
        self.W_value = nn.Linear(d_in, d_out, bias=qkv_bias)
        
        # 2. 定义 Dropout 层
        self.dropout = nn.Dropout(dropout)
        
        # 3. 【关键】注册缓冲区 (Register Buffer)
        # 我们创建了一个 "上三角全为1" 的矩阵，用来做掩码。
        # 为什么要 register_buffer？
        # 因为这个 mask 不是参数(不需要梯度下降)，但它需要随模型一起移动到 GPU 上。
        self.register_buffer(
            'mask',
            torch.triu(torch.ones(context_length, context_length), diagonal=1)
        )

    def forward(self, x):
        # b: Batch Size (批次大小)
        # num_tokens: 当前句子的长度
        b, num_tokens, d_in = x.shape
        
        # 4. 计算 Q, K, V
        keys = self.W_key(x)
        queries = self.W_query(x)
        values = self.W_value(x)
        
        # 5. 计算分数 (Scores)
        # transpose(1, 2) 是处理 [Batch, Seq, Dim] 的标准转置写法
        attn_scores = queries @ keys.transpose(1, 2)
        
        # 6. 【关键】应用因果掩码 (Masking)
        # self.mask.bool()[:num_tokens, :num_tokens] : 截取当前句子长度对应的掩码
        # masked_fill_: 把掩码为 True (右上角) 的地方填成 -inf
        attn_scores.masked_fill_(
            self.mask.bool()[:num_tokens, :num_tokens], 
            -torch.float('inf') # 负无穷
        )
        
        # 7. 归一化 (Softmax)
        # -inf 变成了 0
        attn_weights = torch.softmax(
            attn_scores / keys.shape[-1]**0.5, dim=-1
        )
        
        # 8. 应用 Dropout (随机丢弃)
        attn_weights = self.dropout(attn_weights)
        
        # 9. 加权求和
        context_vec = attn_weights @ values
        
        return context_vec
```

```python
import torch
import torch.nn as nn

class MultiHeadAttentionWrapper(nn.Module):
    def __init__(self, d_in, d_out, context_length, dropout, num_heads, qkv_bias=False):
        super().__init__()
        
        # 1. 创建“专家组” (ModuleList)
        # 我们会有 num_heads 个独立的 CausalAttention 实例
        # 注意：每个头的输出维度必须被 num_heads 整除，这样拼起来才不会变大
        self.heads = nn.ModuleList(
            [CausalAttention(d_in, d_out, context_length, dropout, qkv_bias) 
             for _ in range(num_heads)]
        )
        
        # 2. 输出投影层 (Output Projection)
        # 把大家拼凑起来的结果，再做一次“混合会议”，统一成最终输出
        # 输入维度是 d_out * num_heads (拼接后的长度)
        # 这里的 d_out * num_heads 通常等于原始模型的嵌入维度
        self.out_proj = nn.Linear(d_out * num_heads, d_out * num_heads)

    def forward(self, x):
        # 3. 让每个头独立工作
        # 每个 head(x) 都会返回一个形状为 [Batch, Seq_Len, d_out] 的张量
        context_vec_list = [head(x) for head in self.heads]
        
        # 4. 拼接结果 (Concatenate)
        # 在最后一个维度 (dim=-1) 上拼接
        # 结果形状: [Batch, Seq_Len, d_out * num_heads]
        context_vec = torch.cat(context_vec_list, dim=-1)
        
        # 5. 最终混合
        return self.out_proj(context_vec)

class MultiHeadAttention(nn.Module):
    def __init__(self, d_in, d_out, context_length, dropout, num_heads, qkv_bias=False):
        super().__init__()
        
        # 检查输出维度是否能被头数整除
        assert d_out % num_heads == 0, "d_out must be divisible by num_heads"

        self.d_out = d_out
        self.num_heads = num_heads
        # 计算每个头的维度 (例如: 768 / 12 = 64)
        self.head_dim = d_out // num_heads 

        # 定义 Q, K, V 的线性投影层
        self.W_query = nn.Linear(d_in, d_out, bias=qkv_bias)
        self.W_key = nn.Linear(d_in, d_out, bias=qkv_bias)
        self.W_value = nn.Linear(d_in, d_out, bias=qkv_bias)
        
        # 输出投影层 (用于整合多头的结果)
        self.out_proj = nn.Linear(d_out, d_out)
        
        # Dropout 层
        self.dropout = nn.Dropout(dropout)
        
        # 注册因果掩码缓冲区 (不作为模型参数训练)
        # 创建一个上三角矩阵，对角线以上为 1
        self.register_buffer(
            "mask",
            torch.triu(torch.ones(context_length, context_length), diagonal=1)
        )

    def forward(self, x):
        # x 的形状: [batch_size, num_tokens, d_in]
        b, num_tokens, d_in = x.shape

        # 1. 计算 Q, K, V
        # 形状变为: [batch_size, num_tokens, d_out]
        keys = self.W_key(x)
        queries = self.W_query(x)
        values = self.W_value(x)

        # 2. 隐式拆分多头 (通过 Reshape 和 Transpose)
        # 变换过程:
        # [b, num_tokens, d_out] 
        # -> [b, num_tokens, num_heads, head_dim] (view)
        # -> [b, num_heads, num_tokens, head_dim] (transpose)
        keys = keys.view(b, num_tokens, self.num_heads, self.head_dim)
        values = values.view(b, num_tokens, self.num_heads, self.head_dim)
        queries = queries.view(b, num_tokens, self.num_heads, self.head_dim)

        keys = keys.transpose(1, 2)
        queries = queries.transpose(1, 2)
        values = values.transpose(1, 2)

        # 3. 计算注意力分数 (Scaled Dot-Product Attention)
        # 矩阵乘法在最后两个维度进行: (num_tokens, head_dim) @ (head_dim, num_tokens)
        # 结果形状: [b, num_heads, num_tokens, num_tokens]
        attn_scores = queries @ keys.transpose(2, 3) 

        # 4. 应用因果掩码
        # 将未来位置的分数设为 -inf (负无穷)
        mask_bool = self.mask.bool()[:num_tokens, :num_tokens]
        attn_scores.masked_fill_(mask_bool, -torch.inf)

        # 5. 归一化与 Dropout
        # 缩放系数: 1 / sqrt(head_dim)
        attn_weights = torch.softmax(attn_scores / keys.shape[-1]**0.5, dim=-1)
        attn_weights = self.dropout(attn_weights)

        # 6. 计算上下文向量
        # 形状: [b, num_heads, num_tokens, head_dim]
        context_vec = attn_weights @ values

        # 7. 合并多头 (逆操作)
        # -> [b, num_tokens, num_heads, head_dim] (transpose)
        context_vec = context_vec.transpose(1, 2)
        
        # -> [b, num_tokens, d_out] (contiguous & view)
        # contiguous() 是必要的，因为 transpose 破坏了内存连续性
        context_vec = context_vec.contiguous().view(b, num_tokens, self.d_out)

        # 8. 输出投影 (可选，但标准 GPT 都有)
        context_vec = self.out_proj(context_vec)

        return context_vec
```
